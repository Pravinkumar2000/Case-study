package com.spring;

import org.springframework.stereotype.Service;
@Service("hall")
public class Hall {
	Integer id;
	String hallName,hallOwnerName,ownerMobile,ownerEmailId;
	Integer hallFare;
	public Integer getId() {
		return id;
	}
	public void setId(Integer id) {
		this.id = id;
	}
	public String getHallName() {
		return hallName;
	}
	public void setHallName(String hallName) {
		this.hallName = hallName;
	}
	public String getHallOwnerName() {
		return hallOwnerName;
	}
	public void setHallOwnerName(String hallOwnerName) {
		this.hallOwnerName = hallOwnerName;
	}
	public String getOwnerMobile() {
		return ownerMobile;
	}
	public void setOwnerMobile(String ownerMobile) {
		this.ownerMobile = ownerMobile;
	}
	public String getOwnerEmailId() {
		return ownerEmailId;
	}
	public void setOwnerEmailId(String ownerEmailId) {
		this.ownerEmailId = ownerEmailId;
	}
	public Integer getHallFare() {
		return hallFare;
	}
	public void setHallFare(Integer hallFare) {
		this.hallFare = hallFare;
	}
	public Hall() {}
	public Hall(Integer id, String hallName, String hallOwnerName, String ownerMobile, String ownerEmailId, Integer hallFare) {
		
		this.id = id;
		this.hallName = hallName;
		this.hallOwnerName = hallOwnerName;
		this.ownerMobile = ownerMobile;
		this.ownerEmailId = ownerEmailId;
		this.hallFare = hallFare;
	}
	
	
	
}
